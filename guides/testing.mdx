## Overview

CALZADO J&R uses a comprehensive testing strategy with separate frameworks for backend and frontend:

- **Backend:** pytest + httpx for API testing
- **Frontend:** Vitest + React Testing Library for component testing

Both suites support async testing and provide coverage reporting.

## Backend testing (pytest)

### Test database setup

Create a test fixture in `tests/conftest.py`:

```python
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from sqlalchemy.pool import StaticPool
from fastapi.testclient import TestClient

from app.database import Base
from app.main import app
from app.dependencies import get_db

# In-memory SQLite for fast tests
TEST_DATABASE_URL = "sqlite:///:memory:"

engine = create_engine(
    TEST_DATABASE_URL,
    connect_args={"check_same_thread": False},
    poolclass=StaticPool,
)
TestingSessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)


@pytest.fixture(scope="function")
def db():
    """Create a fresh database for each test."""
    Base.metadata.create_all(bind=engine)
    db = TestingSessionLocal()
    try:
        yield db
    finally:
        db.close()
        Base.metadata.drop_all(bind=engine)


@pytest.fixture(scope="function")
def client(db):
    """Create a test client with database dependency override."""
    def override_get_db():
        try:
            yield db
        finally:
            pass
    
    app.dependency_overrides[get_db] = override_get_db
    with TestClient(app) as test_client:
        yield test_client
    app.dependency_overrides.clear()
```

<Info>
Using SQLite in-memory for tests is significantly faster than PostgreSQL and doesn't require external services.
</Info>

    data = response.json()

    assert data["is_active"] == False  # Pending validation

def test_create_employee_admin_only(client, db):
    """Test only admins can create employee accounts."""
    # Setup roles
    admin_role = Role(name="admin", description="Admin")
    employee_role = Role(name="employee", description="Employee")
    db.add_all([admin_role, employee_role])
    db.commit()
    db.refresh(admin_role)
    
    # Create admin user
    admin = User(
        email="admin@calzadojyr.com",
        full_name="Admin User",
        hashed_password=hash_password("Admin123!"),
        role_id=admin_role.id,

#### Testing models

Create `tests/test_models.py`:

```python
import pytest
from datetime import datetime, timezone, timedelta

from app.models.role import Role
from app.models.user import User
from app.models.password_reset_token import PasswordResetToken
from app.utils.security import hash_password


def test_user_soft_delete(db):
    """Test soft delete preserves user data."""
    role = Role(name="client", description="Client")
    db.add(role)
    db.commit()
    db.refresh(role)
    
    user = User(
        email="test@example.com",
        full_name="Test User",
        hashed_password=hash_password("password"),
        role_id=role.id,
    )
    db.add(user)
    db.commit()
    user_id = user.id
    
    # Soft delete
    user.deleted_at = datetime.now(timezone.utc)
    db.commit()
    
    # Verify user still exists but marked as deleted
    deleted_user = db.query(User).filter(User.id == user_id).first()
    assert deleted_user is not None
    assert deleted_user.deleted_at is not None


def test_password_reset_token_expiry(db):
    """Test token expiration logic."""
    role = Role(name="client", description="Client")
    db.add(role)
    db.commit()
    db.refresh(role)
    
    user = User(
        email="test@example.com",
        full_name="Test User",
        hashed_password=hash_password("password"),
        role_id=role.id,
    )
    db.add(user)
    db.commit()
    
    # Create expired token
    expired_token = PasswordResetToken(
        user_id=user.id,
        token="expired_token_abc123",
        expires_at=datetime.now(timezone.utc) - timedelta(hours=1),
        used=False
    )
    db.add(expired_token)
    db.commit()
    
    # Verify token is expired
    assert expired_token.expires_at < datetime.now(timezone.utc)
```

## Frontend testing (Vitest)

### Dependencies

Frontend testing uses Vitest and React Testing Library. Install dependencies:

```bash
cd fe
npm install -D vitest @testing-library/react @testing-library/jest-dom @testing-library/user-event jsdom
```

### Configuration

Add to `fe/vite.config.ts`:

```typescript
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'

export default defineConfig({
  plugins: [react()],
  test: {
    globals: true,
    environment: 'jsdom',
    setupFiles: './src/test/setup.ts',
    coverage: {
      provider: 'v8',
      reporter: ['text', 'html'],
      exclude: ['node_modules/', 'src/test/'],
    },
  },
})
```

Create `fe/src/test/setup.ts`:

```typescript
import { expect, afterEach } from 'vitest'
import { cleanup } from '@testing-library/react'
import * as matchers from '@testing-library/jest-dom/matchers'

// Extend Vitest matchers with jest-dom
expect.extend(matchers)

// Cleanup after each test
afterEach(() => {
  cleanup()
})
```

## Testing best practices

<AccordionGroup>
  <Accordion title="Test behavior, not implementation">
    Focus on what the user sees and does, not internal state:

    ```typescript
    // Good: tests user interaction
    expect(screen.getByText(/bienvenido/i)).toBeInTheDocument()

    // Bad: tests implementation details
    expect(component.state.isLoggedIn).toBe(true)
    ```
  </Accordion>

  <Accordion title="Use realistic test data">
    Match production data patterns:

    ```python
    # Good: realistic email and name
    user = User(
        email="juan.perez@calzadojyr.com",
        full_name="Juan Pérez",
        occupation="Solador"
    )

    # Bad: generic test data
    user = User(
        email="test@test.com",
        full_name="Test User"
    )
    ```
  </Accordion>

  <Accordion title="Clean up after tests">
    Always reset state between tests:

    ```python
    @pytest.fixture(scope="function")
    def db():
        Base.metadata.create_all(bind=engine)
        db = TestingSessionLocal()
        try:
            yield db
        finally:
            db.close()
            Base.metadata.drop_all(bind=engine)  # Clean up
    ```
  </Accordion>

  <Accordion title="Test error cases">
    Don't just test the happy path:

    ```python
    def test_login_with_invalid_email():
        response = client.post("/api/auth/login", json={
            "email": "notanemail",
            "password": "password"
        })
        assert response.status_code == 422
    ```
  </Accordion>

  <Accordion title="Use factories for complex data">
    Create helper functions for test data:

    ```python
    def create_test_user(db, role_name="client", **kwargs):
        role = db.query(Role).filter(Role.name == role_name).first()
        defaults = {
            "email": "test@example.com",
            "full_name": "Test User",
            "hashed_password": hash_password("password"),
            "role_id": role.id,
        }
        defaults.update(kwargs)
        user = User(**defaults)
        db.add(user)
        db.commit()
        db.refresh(user)
        return user
    ```
  </Accordion>
</AccordionGroup>

## Troubleshooting tests

<Accordion title="Tests fail with database connection errors">
  **Problem:** Tests can't connect to PostgreSQL.

  **Solution:** Use SQLite in-memory for unit tests:

  ```python
  TEST_DATABASE_URL = "sqlite:///:memory:"
  ```

  Or wait for PostgreSQL service in CI:

  ```yaml
  options: >-
    --health-cmd pg_isready
    --health-interval 10s
  ```
</Accordion>

<Accordion title="Async tests not running">
  **Problem:** `RuntimeWarning: coroutine was never awaited`.

  **Solution:** Add `pytest-asyncio` and mark tests:

  ```python
  import pytest

  @pytest.mark.asyncio
  async def test_async_function():
      result = await some_async_function()
      assert result == expected
  ```

  Or enable auto mode in `pytest.ini`:

  ```ini
  [pytest]
  asyncio_mode = auto
  ```
</Accordion>

<Accordion title="Frontend tests fail with 'window is not defined'">
  **Problem:** React tests fail in Node environment.

  **Solution:** Configure jsdom in `vite.config.ts`:

  ```typescript
  export default defineConfig({
    test: {
      environment: 'jsdom',
    },
  })
  ```
</Accordion>
